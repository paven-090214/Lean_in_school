## Lab 1 - 5 실습
logs = {
 "user1": [("/home", 200, "2025-01-20T10:00:00"),
 ("/admin", 500, "2025-01-20T10:10:00"),
 ("/dashboard", 200, "2025-01-20T10:20:00")],
 "user2": [("/home", 200, "2025-01-20T11:00:00"),
 ("/admin", 403, "2025-01-20T11:15:00"),
 ("/settings", 500, "2025-01-20T11:30:00")],
 "user3": [("/home", 200, "2025-01-20T12:00:00"),
 ("/dashboard", 200, "2025-01-20T12:20:00")]
}

totalRequest = {}
for user, request in logs.items() :
    totalRequest[user] = len(request)

print('1. 사용자별 총 요청 수 : ', totalRequest)

errorRequest = {}
for user, request in logs.items() :
    errorCount = 0
    for url, status, time in request :
        if 500 <= status < 600 :
            errorCount += 1
    if errorCount > 0 :
        errorRequest[user] = errorCount

print('2. 500번대 상태 코드 방생 사용자와 횟수 : ', errorRequest)

admin_access = {}
addminCount = 0
for user, request in logs.items() :
    for url, status, time in request :
        if url == '/admin' :
            addminCount += 1

        if(addminCount > 0) :
            admin_access[user] = addminCount
print('특정 URL(/admin)에 접근한 사용자와 횟수 : ', addminCount)
