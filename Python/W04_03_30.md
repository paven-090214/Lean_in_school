```python
    email = input("이메일 주소 : ")
    if email == 'quit':
        break
    else :
        if (email.find('@') != -1) :
            if(email.rfind('@') < email.find('.')) :
                print("유효")
            else :
                print("무효")
        else :
            print("무효")
```

```python
while True :
    url = input("url : ")
    if url == '1' :
        break
    else :
        url = url.replace('https://', '')
        url = url.replace('http://', '')
        url = url.replace('www.', '')
        domain = url.split('/')[0]
        print(f'도메인 : {domain}')
```

```python
text = input("문자열 : ")
while True :
    if text = '1' :
        break
    else :
        text_arr = []
        for char in text :
            text_arr.append(chr(ord(char) + 2))
        print('암호화 문자열 : ',''.join(text_arr))
```

## LAB 2-4
```python
sentences = []
print('여러 문장 입력 : ')
while True :
    sentence = input('문장 입력(공백일 경우 종료) : ')
    if not sentence.strip():
        break
    sentences.append(sentence)
print(f'\n입력된 문장 : {sentences}')
print(f'\n------------------')
for line in sentences :
    words =line.split()
    print('words : ', words)
    words.sort()
    sorted_line = ' '.join(words)
    print('정렬된 문장 : ')
    print(sorted_line)
```
